Private Sub Worksheet_Activate()

'поиск и выделение даты в столбце

Dim strVal As String, strValCell As String

Dim lngRowDate As Long


strVal = Format(Now, "DD.MM.YYYY")

strVal = UCase(strVal)

i = 2

strValCell = UCase(Trim(Cells(i, 1).Value))

Do While strVal <> strValCell

strValCell = UCase(Trim(Cells(i, 1).Value))

i = i + 1

Loop


lngRowDate = i - 1 'w_Find.FnFindRowINCln(strShFind, strDate, 1)

If lngRowDate > 0 Then

Cells(lngRowDate - 1, 1).Select

ActiveCell.Offset(1, 0).Range("A1").Select

' ActiveWindow.SmallScroll Down:=6

End If

End Sub
