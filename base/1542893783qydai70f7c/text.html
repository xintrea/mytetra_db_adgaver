SELECT Year(Дата) As Год, Month(Дата) As Месяц, DatePart('ww',Дата,0,2) AS Неделя, Дата,

Модель, Цвет, Цвет_кузова, Цвет_крыши, Отв_смена, ((Sum(SFP2))) As Prod

FROM PAD_Total PAD_Total

WHERE (((Дата) Between #01/01/2016# And #12/11/2017#)

AND Модель IN ('H79', 'H79N', 'HHA'))

GROUP BY Дата, Модель, Цвет, Цвет_кузова, Цвет_крыши, Отв_смена

HAVING (((Sum(SFP2)))>0)
