Проценты - Сумма налога по сумме и ставке

Function esTax(curSum As Currency, intTax As Integer) As Currency

'es 01.07.2011

'Возвращает сумму налога по сумме и ставке

'--------------------------------------------------------------------

'Например:

' Debug.Print esTax(100, 20) 'вернет : 20

'--------------------------------------------------------------------

On Error GoTo esTaxErr

esTax = CCur(Round(curSum / 100 * intTax, 2))

Exit Function

esTaxErr:

esTax = 0: Err.Clear

End Function
