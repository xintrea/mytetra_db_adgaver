Function fun_EachEndRowInRange(ByVal rRange As Range)

Dim arr As Variant, i As Long

arr = rRange.Value

For i = LBound(arr) To UBound(arr)

If Len(Trim(Application.Clean(arr(i, 1)))) > 0 Then fun_EachEndRowInRange = i

Next i

Erase arr

End Function

'------------------------------------------------------------------------------------------------

Function fun_EachEndClnInRange(ByVal rRange As Range)

Dim arr As Variant, i As Long

arr = rRange.Value

For i = LBound(arr) To UBound(arr, 2)

If Len(Trim(Application.Clean(arr(1, i)))) > 0 Then fun_EachEndClnInRange = i

Next i

Erase arr

End Function
