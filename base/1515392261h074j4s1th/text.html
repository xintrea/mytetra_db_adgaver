Sub reLinkDB()

Dim dbs As Database, tdf As TableDef, i%, j%, s$

Set dbs = CurrentDb

For Each tdf In dbs.TableDefs

    s = tdf.Connect

    If Len(s) > 0 Then

        i = InStr(1, s, "DATABASE=") + 9

        j = InStr(i, s, ".mdb") + 3

        s = Mid(s, i, j)

        If Len(Dir(s)) = 0 Then

            s = InputBox("", "", s)

            If Len(s) = 0 Then

            ElseIf Not Len(Dir(s, vbDirectory)) = 0 Then

                s = ";DATABASE=" + s '+ "; PWD=512512" ' если пароль действительно нужен

               tdf.Connect = s

                tdf.RefreshLink

            End If

        End If

    End If

Next

End Sub
