Sub cmdFolderInfo_Click()

Dim myFileSystemObject As FileSystemObject, curFolder As folder

Set myFileSystemObject = New FileSystemObject

Set curFolder = myFileSystemObject.GetFolder("c:\Java_Dev\")

With curFolder

Debug.Print "Created: " & .DateCreated & vbCrLf

Debug.Print "Last Accessed: " & .DateLastAccessed & vbCrLf

Debug.Print "Last Modified: " & .DateLastModified & vbCrLf

Debug.Print "Size: " & Format(.Size / 1000#, "#0.00") & "KB"

End With


Set myFileSystemObject = Nothing

Set curFolder = Nothing

End Sub
