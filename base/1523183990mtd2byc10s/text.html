Как удалить все файлы из папки (VBA)

'удаляет все файлы из папки

'DirPath - путь к папке без разделителя в конце пути

Sub CleanDir(DirPath As String)

Dim AllFilesMask As String 'шаблон пути всех файлов


AllFilesMask = DirPath & "\*.*"

'проверяем наличие файлов, иначе при удалении возникнет ошибка

If Len(Dir(AllFilesMask)) > 0 Then 'есть хотя бы один файл

Kill AllFilesMask

End If

End If

End Sub
