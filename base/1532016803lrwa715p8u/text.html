Public Sub EchoOff()

' Open the Employees form minimized.

Application.Echo False

DoCmd.Hourglass True

DoCmd.OpenForm "Employees", acNormal

DoCmd.Minimize

Application.Echo True

DoCmd.Hourglass False


End Sub
