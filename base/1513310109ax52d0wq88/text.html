Как вытащить в среду VBA выбранные в списке значения при мультиселектном выделении


Обычно, это делается перебором всех значений и выбором тех, у которых Selected = true.
Пример:

With Me.ListBox1
        For i = 0 To .ListCount - 1
            If .Selected(i) Then
             ' тут код
            End If
        Next
End With

OkeTurel,
Dim varItm As Variant

        With Me.ListBox1
                For Each varItm In .ItemsSelected
                     Debug.Print .ItemData(varItm)
                Next varItm
        End With
