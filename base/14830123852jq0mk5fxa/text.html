Sub поиск_ячейки()

Dim iLastRow As Long

iLastRow = Cells(Rows.Count, ActiveCell.Column).End(xlUp).Row

Cells(iLastRow + 1, ActiveCell.Column).Select

End Sub


'Ситуация 1 - ищем последнюю строку содержащую реальные данные


Sub SearchLastRow1()

LastRow = Cells.Find(What:="*", SearchOrder:=xlRows, _

SearchDirection:=xlPrevious, LookIn:=xlValues).Row

End Sub

'Ситуация 2 - ищем последнюю строку содержащую реальные данные или формулы


Sub SearchLastRow2()

LastRow = Cells.Find(What:="*", SearchOrder:=xlRows, _

SearchDirection:=xlPrevious, LookIn:=xlFormulas).Row

End Sub

'Ситуация 3 - ищем последнюю строку содержащую форматирование


Sub SearchLastRow3()

LastRow = ActiveSheet.UsedRange.Rows.Count

End Sub
