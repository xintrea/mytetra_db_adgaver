Поле со Списком (ComboBox) - Проверка выбрано ли значение (через ListIndex)

Private Sub txtCourse_AfterUpdate()

'Проверка выбрано ли значение ComboBox через ListIndex

If Me!txtCourse.ListIndex = -1 Then 'Значение НЕ ВЫБРАНО

ACourse = 0

Me!txtGroup.Requery

Me!txtGroup.Enabled = False

Else 'Значение ВЫБРАНО

ACourse = Me!txtCourse

Me!txtGroup.Requery

Me!txtGroup.Enabled = True

End If


End Sub
