Function FnSortColumn(ByVal strShName As String, _

ByVal iRowSort As Integer, _

ByVal iClnSort As Integer, _

ByVal strClnSortName As String, _

ByRef iOrderType As Integer)

'iRowSort - строка шапки

'iClnSort - столбец сортировки

'iOrderType - тип сортировки (1-2)

'Dim iClnSort As Integer

'Dim OrderType As XlSortOrder

Dim iFindCell As Range

On Error Resume Next

With ThisWorkbook.Worksheets(strShName)

'определяем диапазон таблицы

iCountCln = .Cells(iRowSort, 256).End(xlToLeft).Column

iCountRow = .Columns(1).Rows(65536).End(xlUp).Row

strAddressAll = Range(.Cells(1, 1), .Cells(iCountRow, iCountCln)).Address

If iClnSort = 0 Then 'если столбец не задан

If Len(strClnSortName) > 0 Then ' если имя столбца задано

'поиск столбца сортировки по имени

Set iFindCell = .Rows(iRowSort).Find(What:=strClnSortName, LookIn:=xlFormulas, LookAt:=xlWhole)

' Set iFindCell = .Columns(iNbColumnFind).Find(What:=strFind, LookIn:=xlFormulas, LookAt:=xlWhole)

If Not iFindCell Is Nothing Then iClnSort = iFindCell.Column

Else

Exit Function 'выход если столбец не определен

End If

End If

'диапазон сортировки в столбце

strAddressSort = Range(.Cells(2, iClnSort), .Cells(iCountRow, iClnSort)).Address

With .Sort

.SortFields.Clear

.SortFields.Add Key:=Range(strAddressSort), SortOn:=xlSortOnValues, Order:=iOrderType, DataOption:=xlSortNormal


.SetRange Range(strAddressAll)

.Header = xlYes

.MatchCase = False

.Orientation = xlTopToBottom

.SortMethod = xlPinYin

.Apply

End With

End With

End Function
