Function FnAddNames(ByVal strShName As String, _

ByVal iRowNames As Integer, _

ByVal iClnNames As Integer, _

ByVal iClnEearhRowCount As Integer, _

ByVal strClnNamesName As String, _

ByRef strNamesName As String, _

ByVal strNamesMessage As String)

'iClnEearhRowCount - столбец поиска количества строк

'iRowNames - строка шапки

'iClnNames - столбец где встраиваем список

'strNamesName - имя списка

'strNamesMessage - всплывающее сообщение (.Comment = "")

'Dim iClnSort As Integer

'Dim OrderType As XlSortOrder

Dim iFindCell As Range

On Error Resume Next

With ThisWorkbook

With .Worksheets(strShName)

If iClnNames = 0 Then 'если столбец не задан

If Len(strClnNamesName) > 0 Then ' если имя столбца задано

'поиск столбца сортировки по имени

Set iFindCell = .Rows(iRowNames).Find(What:=strClnNamesName, LookIn:=xlFormulas, LookAt:=xlWhole)

' Set iFindCell = .Columns(iNbColumnFind).Find(What:=strFind, LookIn:=xlFormulas, LookAt:=xlWhole)

If Not iFindCell Is Nothing Then iClnNames = iFindCell.Column

Else

Exit Function 'выход если столбец не определен

End If

End If

'определяем диапазон таблицы

iCountCln = .Cells(iRowNames, 256).End(xlToLeft).Column

If iClnEearhRowCount = 0 Then iClnEearhRowCount = iClnNames 'если столбец не задан

iCountRow = .Columns(iClnEearhRowCount).Rows(65536).End(xlUp).Row + 1

End With

strNamesSource = Range(Cells(iRowNames + 1, iClnNames), Cells(iCountRow, iClnNames)).Address

On Error Resume Next

.Names(strNamesName).Delete

' .Names.Add Name:=strNamesName, RefersToR1C1:="=List!R12C6:R14C6"

.Names.Add Name:=strNamesName, RefersTo:="=" & strShName & "!" & strNamesSource

' .Names.Add Name:=strNamesName, RefersToR1C1:="=" & strShName & "!R" & iRowNames + 1 & "C" & iClnNames & ":" & "R" & iCountRow & "C" & iClnNames & ""

.Names(strNamesName).Comment = ""

End With

hh = Names(strNamesName).RefersTo

End Function


Function EditNames()

EventsChange False


With ThisWorkbook

'.Sheets("Конфигуратор")

Dim iFindCell As Range

Dim aTempPost(), aTempFIO()

With .Sheets("Матрица ILU")

aTempPost = Range(.Cells(iNbRowStart + 2, 1), .Cells(iNbRowEnd, 3)).Value

aTempFIO = Range(.Cells(iNbRowStart - 1, iNbClnStart), .Cells(iNbRowStart, iNbClnEnd)).Value

End With

'

With .Sheets("Конфигуратор")

' NbCol = Cells(1, 256).End(xlToLeft).Column

iRowEnd = .Columns(1).Rows(65536).End(xlUp).Row + 1

Range(.Cells(2, 1), .Cells(iRowEnd, 3)).Value = ""

iRowTemp = 2

For i = LBound(aTempPost) To UBound(aTempPost)

If Len(aTempPost(i, 3)) > 0 Then

.Cells(iRowTemp, 1).Value = aTempPost(i, 1)

.Cells(iRowTemp, 2).Value = aTempPost(i, 2)

.Cells(iRowTemp, 3).Value = aTempPost(i, 3)

iRowTemp = iRowTemp + 1

End If

Next i

iRowEnd = 0

Erase aTempPost

'FIO

Set iFindCell = .Rows(1).Find(What:="Список операторов", LookIn:=xlFormulas, LookAt:=xlWhole)

' Set iFindCell = .Columns(iNbColumnFind).Find(What:=strFind, LookIn:=xlFormulas, LookAt:=xlWhole)

If Not iFindCell Is Nothing Then

iClnNames = iFindCell.Column

iRowEnd = .Columns(iClnNames).Rows(65536).End(xlUp).Row + 1

Else

Exit Function

End If

If iRowEnd <> 0 Then

Range(.Cells(2, iClnNames - 1), .Cells(iRowEnd, iClnNames)).Value = ""

iRowTemp = 2

For i = LBound(aTempFIO) To UBound(aTempFIO, 2)

If Len(aTempFIO(2, i)) > 0 Then

.Cells(iRowTemp, iClnNames).Value = aTempFIO(2, i)

.Cells(iRowTemp, iClnNames - 1).Value = aTempFIO(1, i)

iRowTemp = iRowTemp + 1

End If

Next i

Erase aTempFIO

End If

End With


End With

EventsChange True

End Function
