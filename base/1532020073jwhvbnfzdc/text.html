Private Sub FromBase_Click()

'Копируем из левого поля в правое

'es - 26.11.2012

'---------------------------------------------------------------------------------------

On Error GoTo FromBase_Click_Err

Me!fldPictureOne.Action = acOLECopy

Me!fldPictureOne = Null

Me!fldPictureTwo.Action = acOLEPaste



FromBase_Click_Bye:

Exit Sub


FromBase_Click_Err:

'MsgBox "Error " & Err.Number & vbCrLf & Err.Description & vbCrLf & _

"in procedure FromBase_Click", vbCritical, "Error!"

Resume FromBase_Click_Bye

End Sub
