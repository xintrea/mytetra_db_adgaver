Текстовое поле (TextBox) - Запрет ввода определённых символов

По материалам: http://www.cyberforum.ru/ms-access/thread2031235.html

На русской раскладке клавиатуры разрешается ввод в поле только букв, поробела и BackSpace, больше ни символов ни цифр, ничего.

Private Sub Название_KeyPress(KeyAscii As Integer)

If Not (KeyAscii >= 192 Or KeyAscii = 32 Or KeyAscii = 8) Then

MsgBox "Ввод цифр и латинских символов в поле запрещен!", vbOKOnly, "Ошибка ввода"

KeyAscii = 0

Else

End If

End Sub
