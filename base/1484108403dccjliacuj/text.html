Function goleador(sstr, sCon2, typ As Integer)

If cn2 Is Nothing Then Set cn2 = CreateObject("ADODB.Connection"): cn2.Open sCon2

 

Sql_S = sstr

cn2.Execute Sql_S

 Set rstData = CreateObject("ADODB.Recordset")

rstData.CursorType = adOpenStatic

rstData.CursorLocation = adUseClient

    rstData.LockType = adLockReadOnly

     

rstData.Open Sql_S, cn2

fk = rstData.Fields.Count

If fk <> 0 Then fk = fk - 1

rk = rstData.RecordCount

 

 

ReDim zagol(fk)

If rk = 0 Then ReDim tabl(rk, fk) Else ReDim tabl(rk - 1, fk)

ReDim vse(rk, fk)

n = 1

        For i = 0 To fk

          vse(0, i) = rstData.Fields(i).Name

          zagol(i) = rstData.Fields(i).Name

        Next

 

        For n = 1 To rk

                For i = 0 To fk

          vse(n, i) = rstData.Fields(i)

          tabl(n - 1, i) = rstData.Fields(i)

        Next

       rstData.MoveNext

     Next

     Select Case typ

        Case 1

        goleador = zagol

        Case 2

        goleador = tabl

        Case 3

        goleador = vse

    End Select

 

 

End Function
