Функция запуска запроса возвращающая кол-во обработанных записей

Public Function esQueryUPDRecords(qName As String) As Long

'es 30.11.03

'Выполняет указанный в аргументе запрос и возвращает кол-во обработанных записей

'--------------------------------------------------------------------

Dim qd As QueryDef

On Error GoTo RunQueryErr

Set qd = CurrentDb.QueryDefs(qName)

qd.Execute

esQueryUPDRecords = qd.RecordsAffected

RunQueryBye:

Set qd = Nothing

Exit Function

RunQueryErr:

esQueryUPDRecords = 0

Err.Clear

End Function
