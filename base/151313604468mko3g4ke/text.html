Sub GroupParent()

Dim lngI As Long

    For lngI = Cells(Rows.Count, 2).End(xlUp).Row To 6 Step -1

        If Rows(lngI).OutlineLevel > Rows(lngI - 1).OutlineLevel Then

            Cells(lngI - 1, 1).Interior.Color = vbMagenta

        End If

         

    Next lngI

End Sub
