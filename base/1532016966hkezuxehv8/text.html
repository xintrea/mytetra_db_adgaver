Библиотечные Ссылки (Referenses) - Удаление всех не обязательных библиотечных ссылок

... в файлах MDB используемых только под данные, например.

Private Sub DelAllReferences()

'es 20.10.2012

'Удаление всех не обязательных библиотечных ссылок

'В данном варианте: Ссылку на "DAO" - ОСТАВЛЯЕМ!

' - Путь: C:\Program Files\Common Files\Microsoft Shared\DAO\dao360.dll

' - Встроенная: False

'--------------------------------------------------------------------------

Dim objRef As Reference

For Each objRef In References

If objRef.BuiltIn = False Then 'Если не обязательна

If objRef.Name <> "DAO" Then 'Ссылку на "DAO" - ОСТАВЛЯЕМ!

References.Remove objRef 'Удаляем ...

End If

End If

Next objRef

Set objRef = Nothing


End Sub
