Sub DelDub()

Dim ColsArr(), i&

With ActiveSheet.ListObjects(1)

   ReDim ColsArr(.ListColumns.Count - 1)

   For i = 1 To .ListColumns.Count

      ColsArr(i - 1) = i

   Next

   .Range.RemoveDuplicates (ColsArr), xlYes

End With

End Sub
